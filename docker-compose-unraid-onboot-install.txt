# append the script to /boot/config/go file on unraid to install docker compose on boot

# Install docker-compose on boot
COMPOSE_VERSION="v2.40.3"
COMPOSE_CACHE="/boot/config/docker-compose-bin"
COMPOSE_BIN="/usr/local/bin/docker-compose"

if [ ! -f "$COMPOSE_CACHE" ]; then
    wget -q "https://github.com/docker/compose/releases/download/${COMPOSE_VERSION}/docker-compose-linux-x86_64" \
        -O "$COMPOSE_CACHE"
fi

cp "$COMPOSE_CACHE" "$COMPOSE_BIN"
chmod +x "$COMPOSE_BIN"
mkdir -p /usr/local/lib/docker/cli-plugins
ln -sf /usr/local/bin/docker-compose /usr/local/lib/docker/cli-plugins/docker-compose
